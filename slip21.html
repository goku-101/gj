import java.util.*; class InvalidDateException extends Exception { InvalidDateException(String msg) { super(msg); } } class MyDate { int day, month, year; void acceptDate(Scanner sc) throws InvalidDateException { System.out.print("Enter day: "); day = sc.nextInt(); System.out.print("Enter month: "); month = sc.nextInt(); System.out.print("Enter year: "); year = sc.nextInt(); if (month < 1 || month > 12 || day < 1 || day > 31) { throw new InvalidDateException("Invalid Date"); } if ((month == 4 || month == 6 || month == 9 || month == 11) && day > 30) { throw new InvalidDateException("Invalid Date"); } if (month == 2) { boolean leap = (year % 4 == 0 && (year % 100 != 0 || year % 400 == 0)); if ((leap && day > 29) || (!leap && day > 28)) { throw new InvalidDateException("Invalid Date"); } } } void displayDate() { System.out.printf("Date: %02d/%02d/%04d\n", day, month, year); } public static void main(String[] args) { Scanner sc = new Scanner(System.in); } public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.print("Enter number of employees: "); int n = sc.nextInt(); sc.nextLine(); Employee[] emp = new Employee[n];
