import java.io.*; import java.util.*; class FileInfo { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.print("Enter file name: "); String filename = sc.nextLine(); File f = new File(filename); System.out.println("File exists: " + f.exists()); if(f.exists()) { System.out.println("Readable: " + f.canRead()); System.out.println("Writable: " + f.canWrite()); System.out.println("Type: " + (f.isFile() ? "File" : f.isDirectory() ? "Directory" : "Other")); System.out.println("Length in bytes: " + f.length()); } } } import javax.swing.*; import java.awt.*; import java.awt.event.*; class SwingTemperatureConverter extends JFrame implements ActionListener { JTextField tCelsius, tFahrenheit; JButton convertC, convertF; SwingTemperatureConverter() { setLayout(new FlowLayout()); add(new JLabel("Celsius:")); tCelsius = new JTextField(10); add(tCelsius); add(new JLabel("Fahrenheit:")); tFahrenheit = new JTextField(10); add(tFahrenheit); convertC = new JButton("C -> F"); convertF = new JButton("F -> C"); add(convertC); add(convertF); convertC.addActionListener(this); convertF.addActionListener(this); setSize(300,150); setVisible(true); setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); } public void actionPerformed(ActionEvent e) { try { if(e.getSource() == convertC) { double c = Double.parseDouble(tCelsius.getText()); double f = (c * 9/5) + 32; tFahrenheit.setText(String.format("%.1f", f)); } else if(e.getSource() == convertF) { double f = Double.parseDouble(tFahrenheit.getText()); double c = (f - 32) * 5/9; tCelsius.setText(String.format("%.1f", c)); } } catch(Exception ex) { JOptionPane.showMessageDialog(this, "Invalid input"); } } public static void main(String[] args) { new SwingTemperatureConverter(); } }
